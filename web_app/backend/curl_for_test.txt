# Here is a sample curl command to upload files to the API endpoint. 
curl is not for Windows, you can use Postman or any other API testing tool for Windows.
Or you can use Restful API client extension in VSCode/ Or Insomnia/ Or API RESTful Django by {url}/api (Debug mode must be on)
# Login first because this is a protected application
curl -X POST http://localhost:8001/api/token/ \
  -H "Content-Type: application/json" \
  -d '{"username": "your_username", "password": "your_password"}'

# Replace <your_access_token> with the token received from the login response.
curl -X POST http://localhost:8001/api/files/ \
  -H "Authorization: Bearer <your_access_token>" \
  -F "files=@/path/to/file1.txt" \
  -F "files=@/path/to/file2.csv"

# sample curl command to get the list of uploaded files
    curl -X POST http://localhost:8001/api/token/ \
    -H "Content-Type: application/json" \
    -d '{"username": "admin", "password": "admin123"}'

curl -X POST http://localhost:8001/api/files/ \
  -H "Authorization: Bearer <your_access_token>" \
  -F "files=@/path/to/file1.txt" \
  -F "files=@/path/to/file2.csv"

# Or assigne the token to a variable and use it in the curl command
# Login
TOKEN=$(curl -s -X POST http://localhost:8001/api/token/ \
  -H "Content-Type: application/json" \
  -d '{"username": "admin", "password": "admin123"}' | jq -r .access)

# Upload files
curl -X POST http://localhost:8001/api/files/ \
  -H "Authorization: Bearer $TOKEN" \
  -F "files=@/path/to/file1.txt" \
  -F "files=@/path/to/file2.csv"

# Get a list of all uploaded files
curl -X GET http://localhost:8001/api/files/ \
  -H "Authorization: Bearer $TOKEN"

if you run this in shell, make sure you have jq installed to parse the JSON response.
CLI on Linux: sudo apt-get install jq
CLI on Windows as: choco install jq (You must have chocolatey installed first and have to run as admin)
